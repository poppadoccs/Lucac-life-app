```jsx
import { useRef, useEffect, useState, useCallback } from "react";
import { SWATCH_COLORS } from "./utils";

// AvatarDrawing — KIDS-04
// Full-screen canvas modal where a kid finger-draws their avatar.
// Saves to Firebase as base64 JPEG at kidsData/{kidName}/avatar via fbSet.
// Per D-15: HTML5 Canvas + Pointer Events API + devicePixelRatio scaling, zero new deps.
// Per D-17: single brush, 5-6 color swatches from SWATCH_COLORS, clear button, save button. No layers, no undo.
//
// Props:
//   V          — theme object
//   kidName    — string, the kid whose avatar is being saved
//   fbSet      — Firebase write helper from App.jsx
//   showToast  — toast helper from App.jsx
//   onClose    — function called when the kid closes the modal (Cancel or Save)
//
// Phase boundary: This component does NOT integrate the saved avatar into the kid header.
// That integration lives in Plan 04, which renders <AvatarDrawing> conditionally and
// reads kidsData[kidName].avatar to display the result in the header.

export default function AvatarDrawing({ V, kidName, fbSet, showToast, onClose }) {
  const canvasRef = useRef(null);
  const isDrawingRef = useRef(false);
  const lastPointRef = useRef(null);
  const [color, setColor] = useState((SWATCH_COLORS && SWATCH_COLORS[0]) || "#111111");
  const [strokeWidth] = useState(8); // single brush, fixed width per D-17
  const [saving, setSaving] = useState(false);

  // Six color swatches from SWATCH_COLORS — D-17 says "5-6 swatches"
  const palette = (SWATCH_COLORS || []).slice(0, 6);

  // Set up canvas with devicePixelRatio scaling for crisp retina rendering
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const dpr = window.devicePixelRatio || 1;
    const rect = canvas.getBoundingClientRect();
    // Set the internal pixel buffer size (high resolution)
    canvas.width = rect.width * dpr;
    canvas.height = rect.height * dpr;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;
    // Scale the context so all drawing uses CSS pixels
    ctx.scale(dpr, dpr);
    ctx.lineCap = "round";
    ctx.lineJoin = "round";
    // Fill background white so saved JPEG isn't transparent (JPEG has no alpha)
    ctx.fillStyle = "#ffffff";
    ctx.fillRect(0, 0, rect.width, rect.height);
  }, []);

  // --- Pointer event handlers (Pointer Events API unifies touch + mouse + stylus) ---
  const getPos = useCallback((e) => {
    const canvas = canvasRef.current;
    if (!canvas) return { x: 0, y: 0 };
    const rect = canvas.getBoundingClientRect();
    return {
      x: e.clientX - rect.left,
      y: e.clientY - rect.top,
    };
  }, []);

  const handlePointerDown = useCallback((e) => {
    // Only respond to primary pointer (prevents multi-touch chaos)
    if (e.button != null && e.button !== 0 && e.pointerType === "mouse") return;
    e.preventDefault();
    const canvas = canvasRef.current;
    if (canvas && canvas.setPointerCapture) {
      try { canvas.setPointerCapture(e.pointerId); } catch (_) { /* ignore */ }
    }
    isDrawingRef.current = true;
    lastPointRef.current = getPos(e);
    // Draw a single dot for tap-to-mark
    const ctx = canvas && canvas.getContext("2d");
    if (ctx) {
      ctx.fillStyle = color;
      ctx.beginPath();
      ctx.arc(lastPointRef.current.x, lastPointRef.current.y, strokeWidth / 2, 0, Math.PI * 2);
      ctx.fill();
    }
  }, [color, strokeWidth, getPos]);

  const handlePointerMove = useCallback((e) => {
    if (!isDrawingRef.current) return;
    e.preventDefault();
    const canvas = canvasRef.current;
    const ctx = canvas && canvas.getContext("2d");
    if (!ctx) return;
    const pos = getPos(e);
    const last = lastPointRef.current || pos;
    ctx.strokeStyle = color;
    ctx.lineWidth = strokeWidth;
    ctx.beginPath();
    ctx.moveTo(last.x, last.y);
    ctx.lineTo(pos.x, pos.y);
    ctx.stroke();
    lastPointRef.current = pos;
  }, [color, strokeWidth, getPos]);

  const handlePointerUp = useCallback((e) => {
    isDrawingRef.current = false;
    lastPointRef.current = null;
    const canvas = canvasRef.current;
    if (canvas && canvas.releasePointerCapture && e.pointerId != null) {
      try { canvas.releasePointerCapture(e.pointerId); } catch (_) { /* ignore */ }
    }
  }, []);

  // --- Clear ---
  function clearCanvas() {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext("2d");
    if (!ctx) return;
    const rect = canvas.getBoundingClientRect();
    ctx.fillStyle = "#ffffff";
    ctx.fillRect(0, 0, rect.width, rect.height);
  }

  // --- Save ---
  // Per D-15: canvas.toDataURL("image/jpeg", 0.5) for size control
  // Per D-16: stored at kidsData/{kidName}/avatar
  function saveAvatar() {
    if (!kidName || !fbSet) {
      showToast && showToast("Cannot save — missing kid name", "error");
      return;
    }
    const canvas = canvasRef.current;
    if (!canvas) return;
    setSaving(true);
    try {
      const dataUrl = canvas.toDataURL("image/jpeg", 0.5);
      // Write to kidsData/{kidName}/avatar — Phase 1 canWrite already permits this path for the kid
      fbSet(`kidsData/${kidName}/avatar`, dataUrl);
      showToast && showToast("Avatar saved!", "success");
      setSaving(false);
      onClose && onClose();
    } catch (err) {
      setSaving(false);
      showToast && showToast("Couldn't save avatar — try again", "error");
    }
  }

  // --- Styles ---
  const overlay = {
    position: "fixed",
    inset: 0,
    background: "rgba(0,0,0,0.85)",
    zIndex: 1000,
    display: "flex",
    flexDirection: "column",
    alignItems: "center",
    justifyContent: "center",
    padding: 16,
  };
  const modal = {
    background: V.bgCard,
    borderRadius: V.r3,
    padding: 16,
    width: "100%",
    maxWidth: 480,
    boxShadow: V.shadowCard,
    display: "flex",
    flexDirection: "column",
    gap: 12,
  };
  const title = {
    fontSize: 20,
    fontWeight: 800,
    color: V.accent,
    textAlign: "center",
    margin: 0,
  };
  const canvasStyle = {
    width: "100%",
    height: 360,
    background: "#ffffff",
    borderRadius: V.r2,
    border: `2px solid ${V.borderDefault}`,
    touchAction: "none", // critical: prevents browser touch gestures (scroll/zoom) from hijacking pointer events
    cursor: "crosshair",
    display: "block",
  };
  const paletteRow = {
    display: "flex",
    gap: 8,
    justifyContent: "center",
    flexWrap: "wrap",
  };
  const swatchBtn = (c, isActive) => ({
    width: 44,
    height: 44,
    minWidth: 44,
    minHeight: 44,
    borderRadius: "50%",
    background: c,
    border: isActive ? `4px solid ${V.accent}` : `2px solid ${V.borderDefault}`,
    cursor: "pointer",
    boxShadow: isActive ? V.shadowCard : "none",
  });
  const buttonRow = {
    display: "flex",
    gap: 8,
    marginTop: 4,
  };
  const btnBase = {
    flex: 1,
    minHeight: 48,
    padding: "10px 14px",
    borderRadius: V.r2,
    border: "none",
    fontSize: 15,
    fontWeight: 700,
    cursor: "pointer",
  };

  return (
    <div style={overlay} role="dialog" aria-label="Draw your avatar">
      <div style={modal}>
        <h2 style={title}>{"\u{1F3A8}"} Draw your face, {kidName}!</h2>
        <canvas
          ref={canvasRef}
          style={canvasStyle}
          onPointerDown={handlePointerDown}
          onPointerMove={handlePointerMove}
          onPointerUp={handlePointerUp}
          onPointerCancel={handlePointerUp}
          onPointerLeave={handlePointerUp}
          aria-label="Drawing canvas"
        />
        <div style={paletteRow} role="radiogroup" aria-label="Color picker">
          {palette.map((c) => (
            <button
              key={c}
              type="button"
              style={swatchBtn(c, color === c)}
              onClick={() => setColor(c)}
              aria-label={`Select color ${c}`}
              aria-pressed={color === c}
            />
          ))}
        </div>
        <div style={buttonRow}>
          <button
            type="button"
            onClick={clearCanvas}
            style={{ ...btnBase, background: V.bgCardAlt, color: V.textSecondary, border: `1px solid ${V.borderDefault}` }}
            aria-label="Clear canvas"
          >
            {"\u{1F5D1} Clear"}
          </button>
          <button
            type="button"
            onClick={onClose}
            style={{ ...btnBase, background: V.bgCardAlt, color: V.textSecondary, border: `1px solid ${V.borderDefault}` }}
            aria-label="Cancel and close"
          >
            {"\u2716 Cancel"}
          </button>
          <button
            type="button"
            onClick={saveAvatar}
            disabled={saving}
            style={{ ...btnBase, background: V.accent, color: "#fff", opacity: saving ? 0.6 : 1 }}
            aria-label="Save avatar"
          >
            {saving ? "Saving..." : "\u2714 Save"}
          </button>
        </div>
      </div>
    </div>
  );
}
```

Key implementation notes:
- `touchAction: "none"` on the canvas style is CRITICAL — without it, mobile browsers will hijack touch events for scroll/zoom and the kid won't be able to draw
- `setPointerCapture` ensures pointer events keep firing on the canvas even if the finger drifts off the edge mid-stroke
- The canvas is initialized with `devicePixelRatio` scaling so the saved drawing is crisp on retina displays
- Background is filled white (not transparent) because JPEG has no alpha channel — without this, the saved image would be black where untouched
- Per D-17, the toolset is intentionally minimal: 6 colors, 1 brush size, clear, cancel, save. No undo, no layers, no fancy brushes (Luca is 6)
- All buttons satisfy mobile-first 44px minimum; save/clear/cancel use 48px for kid friendliness
- ARIA labels and `role="dialog"` make this accessible
- Per `<phase boundary>` in objective: the saved avatar is NOT integrated into the kid header here. Plan 04 reads `kidsData[kidName].avatar` and renders it.
