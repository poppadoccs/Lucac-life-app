```javascript
// ═══ ROLE-BASED EVENT FILTERING ═══
// Filters events based on current user's role:
//   admin  → sees everything
//   parent → sees non-private events + own private events
//   kid    → sees: family events (creator === "admin" || tag === "shared") + own personal events (who === their name) — D-12
//   guest  → sees non-private events (read-only)
function filterEventsForRole(eventsObj, profile) {
  if (!eventsObj || !profile) return {};
  const role = profile.type === "admin" ? "admin"
    : (profile.type === "parent" || profile.type === "family") ? "parent"
    : profile.type === "kid" ? "kid" : "guest";
  if (role === "admin") return eventsObj;
  const filtered = {};
  Object.entries(eventsObj).forEach(([dk, dayEvs]) => {
    const visible = (Array.isArray(dayEvs) ? dayEvs : []).filter(ev => {
      // SEC-02 Phase 1: privacy filter (D-14/D-15/D-16) — stays intact
      const isPrivateEvent = ev.isPrivate ?? ev.private ?? false;
      if (isPrivateEvent) {
        const eventCreator = ev.creator ?? "admin";
        return eventCreator === profile.name;
      }
      // D-12 Phase 2: family-event rule — applies to kids only
      // Legacy events without ev.tag default to null and the rule still works
      if (role === "kid") {
        const tag = ev.tag ?? null;
        const eventCreator = ev.creator ?? "admin";
        const isFamilyEvent = eventCreator === "admin" || tag === "shared";
        const isMyEvent = ev.who && ev.who === profile.name;
        return isFamilyEvent || isMyEvent;
      }
      // Parent and guest: see all non-private events (existing behavior)
      return true;
    });
    if (visible.length > 0) filtered[dk] = visible;
  });
  return filtered;
}
```

```javascript
// ═══ KIDS-02/03/D-10: Hybrid kid routine/goal data model ═══
// DEFINITIONS (admin-controlled): profiles[i].routines and profiles[i].goals
//   Each entry: { id, text, emoji, order }
// COMPLETION STATE (kid-writable):
//   kidsData/{name}/routineState/{dateKey} = { routineId1: true, routineId2: true }
//   kidsData/{name}/goalState/{goalId} = true | false

// Helper: today's ISO date key (YYYY-MM-DD) for routineState lookups
function todayDateKey() {
  const d = new Date();
  return d.getFullYear() + "-" + String(d.getMonth() + 1).padStart(2, "0") + "-" + String(d.getDate()).padStart(2, "0");
}

// Get the routine definitions for a kid (read from their profile, admin-controlled)
function getKidRoutineDefs(kidName) {
  const p = (profiles || []).find(pp => pp.name === kidName && pp.type === "kid");
  return Array.isArray(p?.routines) ? p.routines : [];
}

// Get the goal definitions for a kid
function getKidGoalDefs(kidName) {
  const p = (profiles || []).find(pp => pp.name === kidName && pp.type === "kid");
  return Array.isArray(p?.goals) ? p.goals : [];
}

// Read today's routine completion state for a kid
function getKidRoutineState(kidName) {
  const kd = getKidData(kidName);
  const dk = todayDateKey();
  return (kd.routineState && kd.routineState[dk]) || {};
}

// Read goal completion state for a kid
function getKidGoalState(kidName) {
  const kd = getKidData(kidName);
  return kd.goalState || {};
}

// Toggle a kid routine for today + recompute streak per D-11
function toggleKidRoutine(kidName, routineId) {
  const kd = getKidData(kidName);
  const dk = todayDateKey();
  const stateForDay = (kd.routineState && kd.routineState[dk]) || {};
  const next = !stateForDay[routineId];
  const newDayState = { ...stateForDay, [routineId]: next };
  const newRoutineState = { ...(kd.routineState || {}), [dk]: newDayState };
  const updated = { ...kd, routineState: newRoutineState };
  fbSet(`kidsData/${kidName}`, updated);

  // D-11: streak fires if ALL routines for today are now true
  if (next) {
    const defs = getKidRoutineDefs(kidName);
    const allDone = defs.length > 0 && defs.every(r => newDayState[r.id]);
    if (allDone) {
      const streak = computeKidStreak(kidName, newRoutineState, defs);
      if (streak >= 30) {
        triggerConfetti(document.body, "big");
        showToast(`${kidName} earned a 30-day CROWN! 👑`, "success");
      } else if (streak >= 7) {
        triggerConfetti(document.body, "big");
        showToast(`${kidName} STREAK MASTER! 🔥`, "success");
      } else if (streak >= 3) {
        triggerConfetti(document.body, "small");
      }
    }
  }
}

// Compute consecutive-days streak where ALL routines were done that day (D-11)
function computeKidStreak(kidName, routineStateMap, defs) {
  if (!defs || defs.length === 0) return 0;
  let streak = 0;
  const d = new Date();
  for (let i = 0; i < 365; i++) {
    const dk = d.getFullYear() + "-" + String(d.getMonth() + 1).padStart(2, "0") + "-" + String(d.getDate()).padStart(2, "0");
    const dayState = (routineStateMap && routineStateMap[dk]) || {};
    const allDone = defs.every(r => dayState[r.id]);
    if (allDone) streak++;
    else break;
    d.setDate(d.getDate() - 1);
  }
  return streak;
}

// Toggle a kid goal (persistent until admin removes it)
function toggleKidGoal(kidName, goalId) {
  const kd = getKidData(kidName);
  const next = !(kd.goalState && kd.goalState[goalId]);
  const newGoalState = { ...(kd.goalState || {}), [goalId]: next };
  const updated = { ...kd, goalState: newGoalState };
  fbSet(`kidsData/${kidName}`, updated);
  if (next) triggerConfetti(document.body, "small");
}
```

The streak computation walks back 365 days, breaking on the first day where not-all-routines were done. This is O(streak length) so it's effectively O(1) for the typical 0-30 day case. No new state needed.

Note these helpers all live in the App.jsx component scope so they have closure access to `profiles`, `kidsData`, `fbSet`, `showToast`. They don't need `useCallback` because they're called from JSX click handlers (not passed to memoized children).
